\section*{Primitive Relations}
\begin{longtable}{p{2.5cm}p{8cm}}
    \toprule
    Symbol & Definition \\
    \midrule\endhead
    \texttt{po}   & Program Order -- relates events by their syntactic order within a single thread; a strict total order per thread \\
    \midrule
    \texttt{rf}   & Read-From -- relates a write event to the read event that reads the value it wrote \\
    \midrule
    \texttt{co}   & Coherence Order -- relates a write event to the write event that overwrites it at the same memory location; a strict total order per location \\
    \midrule
    \texttt{rmw}  & Read-Modify-Write -- relates the read and write parts of a successful atomic RMW instruction; the two events are \texttt{po}-adjacent \\
    \midrule
    \texttt{addr} & Address Dependency -- the result of a read is used as the memory address of a subsequent instruction \\
    \midrule
    \texttt{data} & Data Dependency -- the result of a read is used as the data value of a subsequent instruction \\
    \midrule
    \texttt{ctrl} & Control Dependency -- the result of a read determines the control flow leading to a subsequent instruction \\
    \bottomrule
\end{longtable}

\section*{Derived Relations}
\begin{longtable}{p{2.5cm}p{8cm}}
    \toprule
    Symbol & Definition \\
    \midrule\endhead
    \texttt{fr}                      & From-Read -- relates a read to a write that overwrites the value it read; $\mathit{fr} = \mathit{rf}^{-1} \mathbin{;} \mathit{co}$ \\
    \midrule
    \texttt{com}                     & Communication -- $\mathit{com} = \mathit{rf} \cup \mathit{co} \cup \mathit{fr}$ \\
    \midrule
    \texttt{rfe}                     & External Read-From -- \texttt{rf} edges between events on different threads \\
    \midrule
    \texttt{rfi}                     & Internal Read-From -- \texttt{rf} edges between events on the same thread \\
    \midrule
    \texttt{coe}                     & External Coherence Order -- \texttt{co} edges between events on different threads \\
    \midrule
    \texttt{coi}                     & Internal Coherence Order -- \texttt{co} edges between events on the same thread \\
    \midrule
    \texttt{fre}                     & External From-Read -- \texttt{fr} edges between events on different threads \\
    \midrule
    \texttt{fri}                     & Internal From-Read -- \texttt{fr} edges between events on the same thread \\
    \midrule
    \texttt{dep}                     & Dependency -- $\mathit{dep} = \mathit{addr} \cup \mathit{data}$ \\
    \midrule
    \texttt{rwdep}                   & Read-Write Dependency -- $(\mathit{dep} \cup \mathit{ctrl}) \cap (R \times W)$ \\
    \midrule
    \texttt{rrdep}                   & Read-Read Dependency -- $\mathit{addr} \cup (\mathit{dep} \mathbin{;} \mathit{rfi})$ \\
    \midrule
    \texttt{overwrite}               & Overwrite -- $\mathit{overwrite} = \mathit{co} \cup \mathit{fr}$ \\
    \midrule
    \texttt{to-w}             & To-Write ordering -- $\mathit{rwdep} \cup (\mathit{overwrite} \cap \mathit{int})$ \\
    \midrule
    \texttt{to-r}             & To-Read ordering -- $\mathit{strong\text{-}rrdep} \cup \mathit{rfi\text{-}rel\text{-}acq}$ \\
    \midrule
    \texttt{ppo}                     & Preserved Program Order -- $\mathit{rrdep}^{*} \mathbin{;} (\mathit{to\text{-}r} \cup \mathit{to\text{-}w} \cup \mathit{fence})$ \\
    \midrule
    \texttt{fence}                   & Fence -- $\mathit{strong\text{-}fence} \cup \mathit{po\text{-}rel} \cup \mathit{wmb} \cup \mathit{rmb} \cup \mathit{acq\text{-}po}$ \\
    \midrule
    \texttt{strong-fence}     & Strong Fence -- the full memory barrier relation $\mathit{mb}$ \\
    \midrule
    \texttt{cumul-fence}      & Cumulative Fence -- $A\text{-}cumul(\mathit{strong\text{-}fence} \cup \mathit{po\text{-}rel}) \cup \mathit{wmb}$ \\
    \midrule
    \texttt{prop}                    & Propagation -- $(\mathit{overwrite} \cap \mathit{ext})^{?} \mathbin{;} \mathit{cumul\text{-}fence}^{*} \mathbin{;} \mathit{rfe}^{?}$ \\
    \midrule
    \texttt{hb}                      & Happens-Before -- $((\mathit{prop} \setminus \mathit{id}) \cap \mathit{int}) \cup \mathit{ppo} \cup \mathit{rfe}$ \\
    \midrule
    \texttt{pb}                      & Propagates-Before -- $\mathit{prop} \mathbin{;} \mathit{strong\text{-}fence} \mathbin{;} \mathit{hb}^{*}$ \\
    \midrule
    \texttt{xppo}                    & TSO Preserved Program Order -- $((W \times W) \cup (R \times W) \cup (R \times R)) \cap \mathit{po}$ \\
    \midrule
    \texttt{implied}                 & TSO Implied Ordering -- $\mathit{po} \mathbin{;} [\mathit{At} \cup F] \cup [\mathit{At} \cup F] \mathbin{;} \mathit{po}$ \\
    \midrule
    \texttt{At}                      & Atomic events -- $\mathrm{dom}(\mathit{rmw}) \cup \mathrm{codom}(\mathit{rmw})$ \\
    \midrule
    \texttt{A-cumul($r$)}       & A-Cumulative fence -- $\mathit{rfe}^{?} \mathbin{;} r$ \\
    \bottomrule
\end{longtable}

\section*{Mathematical Notation}
\begin{longtable}{p{2.5cm}p{8cm}}
    \toprule
    Symbol & Definition \\
    \midrule\endhead
    $R^{-1}$                  & Inverse of relation $R$: $(a,b) \in R^{-1}$ iff $(b,a) \in R$ \\
    \midrule
    $R \mathbin{;} S$         & Composition of $R$ and $S$: $(a,c)$ iff $\exists b.\ (a,b)\in R \land (b,c)\in S$ \\
    \midrule
    $R \cup S$                & Union of relations (written \texttt{|} in Cat) \\
    \midrule
    $R \cap S$                & Intersection of relations (written \texttt{\&} in Cat) \\
    \midrule
    $R \setminus S$           & Difference of relations \\
    \midrule
    $R^{+}$                   & Transitive closure of $R$ \\
    \midrule
    $R^{*}$                   & Reflexive-transitive closure of $R$ \\
    \midrule
    $R^{?}$                   & Optional relation: $R^{?} = \mathit{id} \cup R$ \\
    \midrule
    $[S]$                     & Identity relation restricted to set $S$: $\{(e,e) \mid e \in S\}$ \\
    \midrule
    $S_1 \times S_2$          & Cartesian product as a relation: $\{(e_1,e_2) \mid e_1 \in S_1 \land e_2 \in S_2\}$ \\
    \midrule
    $\mathrm{dom}(R)$         & Domain of $R$: $\{a \mid \exists b.\ (a,b) \in R\}$ \\
    \midrule
    $\mathrm{codom}(R)$       & Codomain of $R$: $\{b \mid \exists a.\ (a,b) \in R\}$ \\
    \midrule
    \texttt{id}               & Identity relation: $\{(e,e) \mid e \in E\}$ \\
    \midrule
    \texttt{int}              & Internal: pairs of events belonging to the same thread \\
    \midrule
    \texttt{ext}              & External: pairs of events belonging to different threads \\
    \midrule
    \textsc{Acyclic}$(R)$     & No event $a$ satisfies $R^{+}(a,a)$; the relation contains no directed cycle \\
    \midrule
    \textsc{Irreflexive}$(R)$ & No event $a$ satisfies $R(a,a)$ \\
    \midrule
    \textsc{Empty}$(R)$       & $R = \emptyset$ \\
    \bottomrule
\end{longtable}
