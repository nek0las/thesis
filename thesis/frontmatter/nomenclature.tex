\section*{Primitive Relations}
\begin{longtable}{p{2.5cm}p{8cm}}
    \toprule
    Symbol & Definition \\
    \midrule\endhead
    $po$   & Program Order -- relates events by their syntactic order within a single thread; a strict total order per thread \\
    \midrule
    $rf$   & Read-From -- relates a write event to the read event that reads the value it wrote \\
    \midrule
    $co$   & Coherence Order -- relates a write event to the write event that overwrites it at the same memory location; a strict total order per location \\
    \midrule
    $rmw$  & Read-Modify-Write -- relates the read and write parts of a successful atomic RMW instruction; the two events are $po$-adjacent \\
    \midrule
    $addr$ & Address Dependency -- the result of a read is used as the memory address of a subsequent instruction \\
    \midrule
    $data$ & Data Dependency -- the result of a read is used as the data value of a subsequent instruction \\
    \midrule
    $ctrl$ & Control Dependency -- the result of a read determines the control flow leading to a subsequent instruction \\
    \bottomrule
\end{longtable}

\section*{Derived Relations}
\begin{longtable}{p{2.5cm}p{8cm}}
    \toprule
    Symbol & Definition \\
    \midrule\endhead
    $fr$                      & From-Read -- relates a read to a write that overwrites the value it read; $fr = rf^{-1} \mathbin{;} co$ \\
    \midrule
    $com$                     & Communication -- $com = rf \cup co \cup fr$ \\
    \midrule
    $rfe$                     & External Read-From -- $rf$ edges between events on different threads \\
    \midrule
    $rfi$                     & Internal Read-From -- $rf$ edges between events on the same thread \\
    \midrule
    $coe$                     & External Coherence Order -- $co$ edges between events on different threads \\
    \midrule
    $coi$                     & Internal Coherence Order -- $co$ edges between events on the same thread \\
    \midrule
    $fre$                     & External From-Read -- $fr$ edges between events on different threads \\
    \midrule
    $fri$                     & Internal From-Read -- $fr$ edges between events on the same thread \\
    \midrule
    $dep$                     & Dependency -- $dep = addr \cup data$ \\
    \midrule
    $rwdep$                   & Read-Write Dependency -- $(dep \cup ctrl) \cap (R \times W)$ \\
    \midrule
    $rrdep$                   & Read-Read Dependency -- $addr \cup (dep \mathbin{;} rfi)$ \\
    \midrule
    $overwrite$               & Overwrite -- $overwrite = co \cup fr$ \\
    \midrule
    $to\text{-}w$             & To-Write ordering -- $rwdep \cup (overwrite \cap int)$ \\
    \midrule
    $to\text{-}r$             & To-Read ordering -- $strong\text{-}rrdep \cup rfi\text{-}rel\text{-}acq$ \\
    \midrule
    $ppo$                     & Preserved Program Order -- $rrdep^{*} \mathbin{;} (to\text{-}r \cup to\text{-}w \cup fence)$ \\
    \midrule
    $fence$                   & Fence -- $strong\text{-}fence \cup po\text{-}rel \cup wmb \cup rmb \cup acq\text{-}po$ \\
    \midrule
    $strong\text{-}fence$     & Strong Fence -- the full memory barrier relation $mb$ \\
    \midrule
    $cumul\text{-}fence$      & Cumulative Fence -- $A\text{-}cumul(strong\text{-}fence \cup po\text{-}rel) \cup wmb$ \\
    \midrule
    $prop$                    & Propagation -- $(overwrite \cap ext)^{?} \mathbin{;} cumul\text{-}fence^{*} \mathbin{;} rfe^{?}$ \\
    \midrule
    $hb$                      & Happens-Before -- $((prop \setminus id) \cap int) \cup ppo \cup rfe$ \\
    \midrule
    $pb$                      & Propagates-Before -- $prop \mathbin{;} strong\text{-}fence \mathbin{;} hb^{*}$ \\
    \midrule
    $xppo$                    & TSO Preserved Program Order -- $((W \times W) \cup (R \times W) \cup (R \times R)) \cap po$ \\
    \midrule
    $implied$                 & TSO Implied Ordering -- $po \mathbin{;} [At \cup F] \cup [At \cup F] \mathbin{;} po$ \\
    \midrule
    $At$                      & Atomic events -- $dom(rmw) \cup codom(rmw)$ \\
    \midrule
    $A\text{-}cumul(r)$       & A-Cumulative fence -- $rfe^{?} \mathbin{;} r$ \\
    \bottomrule
\end{longtable}

\section*{Mathematical Notation}
\begin{longtable}{p{2.5cm}p{8cm}}
    \toprule
    Symbol & Definition \\
    \midrule\endhead
    $R^{-1}$                  & Inverse of relation $R$: $(a,b) \in R^{-1}$ iff $(b,a) \in R$ \\
    \midrule
    $R \mathbin{;} S$         & Composition of $R$ and $S$: $(a,c)$ iff $\exists b.\ (a,b)\in R \land (b,c)\in S$ \\
    \midrule
    $R \cup S$                & Union of relations (written \texttt{|} in Cat) \\
    \midrule
    $R \cap S$                & Intersection of relations (written \texttt{\&} in Cat) \\
    \midrule
    $R \setminus S$           & Difference of relations \\
    \midrule
    $R^{+}$                   & Transitive closure of $R$ \\
    \midrule
    $R^{*}$                   & Reflexive-transitive closure of $R$ \\
    \midrule
    $R^{?}$                   & Optional relation: $R^{?} = id \cup R$ \\
    \midrule
    $[S]$                     & Identity relation restricted to set $S$: $\{(e,e) \mid e \in S\}$ \\
    \midrule
    $S_1 \times S_2$          & Cartesian product as a relation: $\{(e_1,e_2) \mid e_1 \in S_1 \land e_2 \in S_2\}$ \\
    \midrule
    $dom(R)$                  & Domain of $R$: $\{a \mid \exists b.\ (a,b) \in R\}$ \\
    \midrule
    $codom(R)$                & Codomain of $R$: $\{b \mid \exists a.\ (a,b) \in R\}$ \\
    \midrule
    $id$                      & Identity relation: $\{(e,e) \mid e \in E\}$ \\
    \midrule
    $int$                     & Internal: pairs of events belonging to the same thread \\
    \midrule
    $ext$                     & External: pairs of events belonging to different threads \\
    \midrule
    \textsc{Acyclic}$(R)$     & No event $a$ satisfies $R^{+}(a,a)$; the relation contains no directed cycle \\
    \midrule
    \textsc{Irreflexive}$(R)$ & No event $a$ satisfies $R(a,a)$ \\
    \midrule
    \textsc{Empty}$(R)$       & $R = \emptyset$ \\
    \bottomrule
\end{longtable}
